[PS1] SOLUSI: Count & Sum Loop

public static void analyzeNumbers(int[] numbers) {
    int count = 0;
    int sum = 0;
    int max = numbers[0];
    int min = numbers[0];
    
    for (int num : numbers) {
        if (num > 5) count++;
        sum += num;
        if (num > max) max = num;
        if (num < min) min = num;
    }
    
    double average = (double) sum / numbers.length;
    
    System.out.println("Angka > 5: " + count + ", Sum: " + sum + 
                     ", Max: " + max + ", Min: " + min + 
                     ", Average: " + String.format("%.2f", average));
}

Output:
Angka > 5: 4, Sum: 35, Max: 9, Min: 1, Average: 5.00

interface Shape {
    double calculateArea();
}

class Circle implements Shape {
    double radius;
    Circle(double r) { this.radius = r; }
    
    @Override
    public double calculateArea() {
        return Math.PI * radius * radius;
    }
}

class Rectangle implements Shape {
    double length, width;
    Rectangle(double l, double w) { this.length = l; this.width = w; }
    
    @Override
    public double calculateArea() {
        return length * width;
    }
}

// Main logic
ArrayList<Shape> shapes = new ArrayList<>();
shapes.add(new Circle(5));
shapes.add(new Rectangle(4, 6));
// ... add more

// a) Display areas
double totalArea = 0;
Shape maxShape = null;
double maxArea = 0;

for (Shape shape : shapes) {
    double area = shape.calculateArea();
    System.out.println("Area: " + area);
    totalArea += area;
    
    if (area > maxArea) {
        maxArea = area;
        maxShape = shape;
    }
}

System.out.println("Total Area: " + totalArea);
System.out.println("Max Area: " + maxArea);

double saldo = 1_000_000;
double withdrawal;
boolean validInput;

do {
    validInput = true;
    System.out.print("Masukkan nominal (kelipatan 50rb): Rp ");
    withdrawal = scanner.nextDouble();
    
    if (withdrawal <= 0) {
        System.out.println("Invalid - harus > 0");
        validInput = false;
    } else if (withdrawal > saldo) {
        System.out.println("Invalid - saldo tidak cukup");
        validInput = false;
    } else if (withdrawal % 50_000 != 0) {
        System.out.println("Invalid - harus kelipatan 50rb");
        validInput = false;
    }
} while (!validInput);

saldo -= withdrawal;
System.out.println("Withdrawal success. Remaining: Rp " + saldo);

public static Product searchByName(ArrayList<Product> products, String name) {
    for (Product p : products) {
        if (p.getName().equalsIgnoreCase(name)) {
            return p;
        }
    }
    return null;
}

public static ArrayList<Product> searchByPriceRange(
        ArrayList<Product> products, double min, double max) {
    ArrayList<Product> result = new ArrayList<>();
    for (Product p : products) {
        if (p.getPrice() >= min && p.getPrice() <= max) {
            result.add(p);
        }
    }
    return result;
}

public static double getAveragePrice(ArrayList<Product> products, String category) {
    double sum = 0;
    int count = 0;
    for (Product p : products) {
        if (p.getCategory().equalsIgnoreCase(category)) {
            sum += p.getPrice();
            count++;
        }
    }
    return count > 0 ? sum / count : 0;
}


[PS7] SOLUSI: Nested Loop Multiplication Table

public static void printMultiplicationTable(int maxNumber) {
    for (int i = 1; i <= maxNumber; i++) {
        for (int j = 1; j <= maxNumber; j++) {
            System.out.print(String.format("%2dx%-2d=%-4d  ", i, j, i*j));
        }
        System.out.println();
    }
}


[PS8] SOLUSI: Polymorphic Employee

abstract class EmployeeBonus {
    String name;
    double baseSalary;
    abstract double calculateBonus();
}

class Manager extends EmployeeBonus {
    @Override
    double calculateBonus() {
        return baseSalary * 0.20;
    }
}

class Developer extends EmployeeBonus {
    @Override
    double calculateBonus() {
        return baseSalary * 0.15;
    }
}

class Designer extends EmployeeBonus {
    @Override
    double calculateBonus() {
        return baseSalary * 0.12;
    }
}

// Main
ArrayList<EmployeeBonus> employees = new ArrayList<>();
// ... add mixed employees

double maxBonus = 0;
EmployeeBonus maxBonusEmp = null;
double totalBonus = 0;

for (EmployeeBonus emp : employees) {
    double bonus = emp.calculateBonus();
    System.out.println(emp.name + " - Rp" + emp.baseSalary + 
                     " - Bonus: Rp" + bonus);
    totalBonus += bonus;
    
    if (bonus > maxBonus) {
        maxBonus = bonus;
        maxBonusEmp = emp;
    }
}

System.out.println("Total Bonus: Rp" + totalBonus);
System.out.println("Max Bonus: " + maxBonusEmp.name + " - Rp" + maxBonus);
